function getCurrentTime() { // Створюємо новий об'єкт Date, який автоматично отримує поточну дату та час const now = new Date(); // Отримуємо години, хвилини та секунди з поточного часу let hours = now.getHours(); let minutes = now.getMinutes(); let seconds = now.getSeconds(); // Додаємо нулі спереду, якщо число менше 10 (для формату HH:MM:SS) hours = hours < 10 ? '0' + hours : hours; minutes = minutes < 10 ? '0' + minutes : minutes; seconds = seconds < 10 ? '0' + seconds : seconds; // Отримуємо назву дня тижня (від 0 до 6, де 0 - неділя, 6 - субота) const daysOfWeek = ['неділя', 'понеділок', 'вівторок', 'середа', 'четвер', 'п’ятниця', 'субота']; const dayOfWeek = daysOfWeek[now.getDay()]; // Отримуємо день місяця, місяця та рік const dayOfMonth = now.getDate(); const month = now.getMonth(); // Від 0 до 11, де 0 - січень, 11 - грудень const year = now.getFullYear(); // Масив з назвами місяців (від 0 до 11) const months = ['січня', 'лютого', 'березня', 'квітня', 'травня', 'червня', 'липня', 'серпня', 'вересня', 'жовтня', 'листопада', 'грудня']; const monthName = months[month]; // Формуємо рядок для виведення відформатованої дати та часу const formattedTime = `${hours}:${minutes}:${seconds}, ${dayOfWeek}, ${dayOfMonth} ${monthName} ${year} року`; // Виводимо результат у консоль console.log(formattedTime); // Повертаємо відформатований час return formattedTime; } // Викликаємо функцію для отримання і виведення поточного часу getCurrentTime();
